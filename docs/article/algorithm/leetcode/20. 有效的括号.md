# [20. 有效的括号](https://leetcode-cn.com/problems/valid-parentheses/)

```js
/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
  const map = {
    ']': '[',
    '}': '{',
    ')': '('
  }
  const stack = []
  for (let i = 0;i < s.length;i++) {
    // 如果是左括号 添加到栈中
    if (!map[s[i]]) {
      stack.push(s[i])
    } else {
      // 如果是右括号，把栈顶元素弹出，并配对
      const t = stack.pop()
      // 括号没有配对成功 返回false
      if (map[s[i]] !== t) {
        return false
      }
    }
  }
  return !stack.length
};
```
