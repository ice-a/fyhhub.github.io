# 53. 最大子序和
```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxSubArray = function(nums) {
  const dp = []
  dp[0] = nums[0]
  let max = -Number.MAX_VALUE
  for (let i = 1;i < nums.length;i++) {
    dp[i] = Math.max(nums[i], dp[i - 1] + nums[i])
    max = Math.max(max, dp[i])
  }
  return Math.max(max, dp[0])
};
```
