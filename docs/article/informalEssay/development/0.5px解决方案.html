# 0.5px解决方案

export default function App() {
  return (
    <div>
      <div className="setOnePx">一条边0.5px</div>
      <div className="setBorderAll">四条边0.5px</div>
    </div>
  )
}
App.__style__ = `
  
  /* =============== 一条边*/
  .setOnePx {
    position: relative;
  }
  .setOnePx::after{
    position: absolute;
    content: '';
    background-color: #e5e5e5;
    display: block;
    width: 100%;
    height: 1px; /*no*/
    transform: scale(1, 0.5);
    top: 0;
    left: 0;
  }
  /* =============== 四条边*/
  .setBorderAll{
    position: relative;
    width: 100px;
    height: 100px;
    margin-top: 32px;
    border-radius: 20px;
  }	
  .setBorderAll::after{
    content:"";
    position:absolute;
    top: 0;
    left: 0;
    width: 200%;
    height: 200%;
    transform: scale(0.5);
    transform-origin: left top;
    box-sizing: border-box;
    border: 1px solid #E5E5E5;
    border-radius: 20px;
  }
`
